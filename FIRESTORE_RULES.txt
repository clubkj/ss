// Copy-paste these rules into Firebase Console → Firestore Database → Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // BOOKINGS - Anyone can create, authenticated users can read/update
    match /bookings/{bookingId} {
      allow create: if true;
      allow read: if true;
      allow update: if request.auth != null;
      allow delete: if false;
    }

    // DRIVERS - Drivers can create/update their own profile, anyone can read
    match /drivers/{driverId} {
      allow read: if true;
      allow create: if request.auth.uid == driverId;
      allow update: if request.auth.uid == driverId;
      allow delete: if false;
    }

    // CUSTOMERS - Customers can create/update their own profile, anyone can read
    match /customers/{customerId} {
      allow read: if true;
      allow create: if request.auth.uid == customerId;
      allow update: if request.auth.uid == customerId;
      allow delete: if false;
    }

    // NOTIFICATIONS - Anyone can create/read
    match /notifications/{notificationId} {
      allow create: if true;
      allow read: if true;
      allow write: if request.auth != null;
      allow delete: if false;
    }
  }
}
